packages = ["m3u8", "ssl", "certifi"] import asyncio import m3u8 import urllib.parse from js import window, document, Uint8Array from pyodide.http import pyfetch import json # Settings CORS_PROXY = "https://corsproxy.io/?url=" def log(msg, type=None): window.addLog(msg, type) async def start_analysis(event): btn = document.getElementById("btn-run") btn.disabled = True document.getElementById("loader").style.display = "inline-block" document.getElementById("status-text").innerText = "분석 중..." document.getElementById("progress-container").style.display = "block" document.getElementById("output-container").innerHTML = "" window.updateProgress(0) url = document.getElementById("m3u8-url").value max_segs = int(document.getElementById("max-segs").value) use_proxy = document.getElementById("use-proxy").checked try: await check_hls_continuity(url, max_segs, use_proxy) except Exception as e: log(f"실행 오류: {str(e)}", "error") finally: btn.disabled = False document.getElementById("loader").style.display = "none" document.getElementById("status-text").innerText = "완료" async def fetch_bytes(url, use_proxy): final_url = f"https://api.allorigins.win/raw?url={urllib.parse.quote(url)}" if use_proxy else url response = await pyfetch(final_url) if not response.ok: raise Exception(f"데이터 로드 실패 (Status: {response.status})") return await response.bytes() async def check_hls_continuity(m3u8_url, max_segments, use_proxy): log(f"[*] M3U8 분석 시작: {m3u8_url}", "header") try: # Load M3U8 content proxy_prefix = "https://api.allorigins.win/raw?url=" if use_proxy else "" content_response = await pyfetch(proxy_prefix + urllib.parse.quote(m3u8_url)) content = await content_response.string() playlist = m3u8.loads(content, uri=m3u8_url) except Exception as e: log(f"[!] M3U8 로드 실패: {e}", "error") return if playlist.is_variant: log("[*] 마스터 플레이리스트 감지됨. 첫 번째 하위 스트림으로 이동합니다.", "info") media_url = playlist.playlists[0].absolute_uri content_response = await pyfetch(CORS_PROXY + urllib.parse.quote(media_url) if use_proxy else media_url) content = await content_response.string() playlist = m3u8.loads(content, uri=media_url) segments = playlist.segments if not segments: log("[!] 세그먼트를 찾을 수 없습니다.", "error") return check_count = min(len(segments), max_segments) log(f"[*] 총 {len(segments)}개의 세그먼트 중 처음 {check_count}개를 검사합니다.", "info") prev_last_pts = None for i in range(check_count): segment = segments[i] seg_url = segment.absolute_uri log(f"--- Segment {i} 분석 중 ---", "header") try: # Fetch segment bytes seg_bytes = await fetch_bytes(seg_url, use_proxy) # Use JS mux.js helper # convert bytes to Uint8Array for JS js_arr = Uint8Array.new(seg_bytes) ts_info = await window.parseSegmentTimestamps(js_arr.buffer) # ts_info is a JS Object {first: ..., last: ...} first_pts = ts_info.first last_pts = ts_info.last if first_pts is None: log(" [경고] 타임스탬프를 찾을 수 없습니다. (비디오 트랙 없음)", "warning") continue log(f" 시작 PTS: {first_pts:.4f} / 종료 PTS: {last_pts if last_pts else '?'}", "info") if last_pts: log(f" 재생 길이: {last_pts - first_pts:.4f}s", "info") # Continuity Check if prev_last_pts is not None: gap = first_pts - prev_last_pts if gap > 0.1: log(f" [오류 후보] 🚨 이전 세그먼트와 시간이 뚝 끊겼습니다! (Gap: +{gap:.4f}초)", "error") elif gap < -0.1: log(f" [오류 후보] 🚨 이전 세그먼트와 시간이 겹칩니다! (Overlap: {gap:.4f}초)", "error") else: log(f" [정상] 이전 세그먼트와 부드럽게 이어집니다. (차이: {gap:.4f}초)", "success") prev_last_pts = last_pts window.updateProgress(int((i+1) / check_count * 100)) except Exception as e: log(f" [에러] 세그먼트 분석 실패: {str(e)}", "error") log("\n[완료] 검사가 완료되었습니다.", "header") # Final boot signal document.getElementById("status-text").innerText = "준비 완료" document.getElementById("output-container").innerHTML = '
